# 差分算法

【差分算法】(Difference Array)是一种用于处理区间修改问题的重要技巧。它可以将O(n)的区间修改操作优化到O(1)，特别适用于有大量区间修改操作的场景。

## 基本概念

差分数组是原数组相邻元素的差值组成的数组。对于原数组`arr[]`，其差分数组`diff[]`定义为：
```
diff[0] = arr[0]
diff[i] = arr[i] - arr[i-1]  (i > 0)
```

通过差分数组可以还原原数组：
```
arr[0] = diff[0]
arr[i] = arr[i-1] + diff[i] = diff[0] + diff[1] + ... + diff[i]
```

**核心思想**：要对区间`[l, r]`的所有元素加上`val`，只需要：
- `diff[l] += val`
- `diff[r+1] -= val`

## 一维差分

### 基本实现

```cpp
#include <iostream>
#include <vector>
using namespace std;

class DifferenceArray {
private:
    vector<long long> diff;
    int n;
    
public:
    // 构造函数，初始化差分数组
    DifferenceArray(const vector<int>& arr) {
        n = arr.size();
        diff.resize(n + 1, 0);
        
        // 构建差分数组
        diff[0] = arr[0];
        for (int i = 1; i < n; i++) {
            diff[i] = arr[i] - arr[i-1];
        }
    }
    
    // 对区间[l, r]的所有元素加上val
    void rangeAdd(int l, int r, int val) {
        diff[l] += val;
        if (r + 1 < n) {
            diff[r + 1] -= val;
        }
    }
    
    // 还原原数组
    vector<long long> getArray() {
        vector<long long> result(n);
        result[0] = diff[0];
        
        for (int i = 1; i < n; i++) {
            result[i] = result[i-1] + diff[i];
        }
        
        return result;
    }
    
    // 获取单个位置的值
    long long getValue(int idx) {
        long long sum = 0;
        for (int i = 0; i <= idx; i++) {
            sum += diff[i];
        }
        return sum;
    }
};

int main() {
    vector<int> arr = {1, 3, 5, 7, 9};
    DifferenceArray da(arr);
    
    cout << "原数组: ";
    auto original = da.getArray();
    for (int x : original) cout << x << " ";
    cout << endl;
    
    // 对区间[1, 3]的元素都加上2
    da.rangeAdd(1, 3, 2);
    
    cout << "修改后: ";
    auto modified = da.getArray();
    for (int x : modified) cout << x << " ";
    cout << endl;
    
    return 0;
}
```

```cpp
class DifferenceArrayFromZero {

    int n;
    
public:

        diff.resize(n + 1, 0);

        if (r + 1 < n) {
            diff[r + 1] -= val;
        }

    // 获取最终数组
    vector<long long> getArray() {
        vector<long long> result(n, 0);
        if (n > 0) {
            result[0] = diff[0];
            for (int i = 1; i < n; i++) {
                result[i] = result[i-1] + diff[i];
            }
        }
        return result;
    }
};
```

## 二维差分

二维差分用于处理矩阵的区间修改问题，可以在O(1)时间内对任意子矩阵的所有元素进行相同的修改。

### 基本原理

对于矩阵中的子矩阵`(x1,y1)`到`(x2,y2)`，要对所有元素加上`val`，需要进行以下操作：
```
diff[x1][y1] += val
diff[x1][y2+1] -= val
diff[x2+1][y1] -= val
diff[x2+1][y2+1] += val
```

### 实现代码

```cpp
#include <iostream>
#include <vector>
using namespace std;

class DifferenceArray2D {
private:
    vector<vector<long long>> diff;
    int rows, cols;
    
public:
    // 从原矩阵构造差分数组
    DifferenceArray2D(const vector<vector<int>>& matrix) {
        rows = matrix.size();
        cols = matrix[0].size();
        diff.assign(rows + 1, vector<long long>(cols + 1, 0));
        
        // 构建二维差分数组
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                rangeAdd(i, j, i, j, matrix[i][j]);
            }
        }
    }
    
    // 从零开始构造
    DifferenceArray2D(int r, int c) : rows(r), cols(c) {
        diff.assign(rows + 1, vector<long long>(cols + 1, 0));
    }
    
    // 对子矩阵(x1,y1)到(x2,y2)的所有元素加上val
    void rangeAdd(int x1, int y1, int x2, int y2, int val) {
        diff[x1][y1] += val;
        diff[x1][y2 + 1] -= val;
        diff[x2 + 1][y1] -= val;
        diff[x2 + 1][y2 + 1] += val;
    }
    
    // 还原原矩阵
    vector<vector<long long>> getMatrix() {
        vector<vector<long long>> result(rows, vector<long long>(cols));
        
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                result[i][j] = diff[i][j];
                if (i > 0) result[i][j] += result[i-1][j];
                if (j > 0) result[i][j] += result[i][j-1];
                if (i > 0 && j > 0) result[i][j] -= result[i-1][j-1];
            }
        }
        
        return result;
    }
};

int main() {
    // 创建3x3的零矩阵
    DifferenceArray2D da2d(3, 3);
    
    // 对子矩阵(0,0)到(1,1)的元素都加5
    da2d.rangeAdd(0, 0, 1, 1, 5);
    
    // 对子矩阵(1,1)到(2,2)的元素都加3
    da2d.rangeAdd(1, 1, 2, 2, 3);
    
    auto matrix = da2d.getMatrix();
    cout << "修改后的矩阵:" << endl;
    for (const auto& row : matrix) {
        for (int val : row) {
            cout << val << " ";
        }
        cout << endl;
    }
    
    return 0;
}
```

## 高级应用

### 1. 差分+前缀和

结合差分和前缀和可以同时处理区间修改和区间查询：

```cpp
class RangeUpdateQuery {
private:
    vector<long long> diff;
    int n;
    
public:
    RangeUpdateQuery(int size) : n(size) {
        diff.resize(n + 1, 0);
    }
    
    // 区间[l, r]加上val
    void rangeAdd(int l, int r, int val) {
        diff[l] += val;
        if (r + 1 < n) {
            diff[r + 1] -= val;
        }
    }
    
    // 计算前缀和数组（在所有修改完成后调用）
    vector<long long> getPrefixSum() {
        vector<long long> arr(n), prefix(n + 1, 0);
        
        // 还原原数组
        arr[0] = diff[0];
        for (int i = 1; i < n; i++) {
            arr[i] = arr[i-1] + diff[i];
        }
        
        // 计算前缀和
        for (int i = 0; i < n; i++) {
            prefix[i + 1] = prefix[i] + arr[i];
        }
        
        return prefix;
    }
    
    // 区间查询（需要先调用getPrefixSum）
    long long rangeQuery(const vector<long long>& prefix, int l, int r) {
        return prefix[r + 1] - prefix[l];
    }
};
```

### 2. 树上差分

用于处理树上路径的修改问题：

```cpp
const int MAXN = 100005;
vector<int> adj[MAXN];
int parent[MAXN], depth[MAXN];
long long diff[MAXN];

// DFS预处理
void dfs(int u, int p, int d) {
    parent[u] = p;
    depth[u] = d;
    
    for (int v : adj[u]) {
        if (v != p) {
            dfs(v, u, d + 1);
        }
    }
}

// 找LCA（简单版本）
int lca(int u, int v) {
    while (depth[u] > depth[v]) u = parent[u];
    while (depth[v] > depth[u]) v = parent[v];
    while (u != v) {
        u = parent[u];
        v = parent[v];
    }
    return u;
}

// 路径u到v加上val
void pathAdd(int u, int v, int val) {
    int l = lca(u, v);
    diff[u] += val;
    diff[v] += val;
    diff[l] -= val;
    if (parent[l] != -1) {
        diff[parent[l]] -= val;
    }
}

// DFS计算最终值
void calculate(int u, int p) {
    for (int v : adj[u]) {
        if (v != p) {
            calculate(v, u);
            diff[u] += diff[v];
        }
    }
}
```

### 3. 环形数组差分

处理环形数组的区间修改：

```cpp
class CircularDifference {
private:
    vector<long long> diff;
    int n;
    
public:
    CircularDifference(int size) : n(size) {
        diff.resize(n, 0);
    }
    
    // 环形区间[l, r]加上val
    void rangeAdd(int l, int r, int val) {
        if (l <= r) {
            // 正常区间
            diff[l] += val;
            diff[(r + 1) % n] -= val;
        } else {
            // 跨越边界的区间
            diff[l] += val;
            diff[0] -= val;
            diff[0] += val;
            diff[(r + 1) % n] -= val;
        }
    }
    
    // 还原数组
    vector<long long> getArray() {
        vector<long long> result(n);
        result[0] = diff[0];
        
        for (int i = 1; i < n; i++) {
            result[i] = result[i-1] + diff[i];
        }
        
        return result;
    }
};
```

### 例题2：拼车问题

**问题描述**：有一辆车，容量为capacity。给定乘客上下车信息，判断车是否会超载。

```cpp
bool carPooling(vector<vector<int>>& trips, int capacity) {
    const int MAXPOS = 1001;
    vector<int> diff(MAXPOS, 0);
    
    // 处理所有行程
    for (const auto& trip : trips) {
        int passengers = trip[0];
        int start = trip[1];
        int end = trip[2];
        
        diff[start] += passengers;
        diff[end] -= passengers;
    }
    
    // 检查每个位置的乘客数
    int currentPassengers = 0;
    for (int i = 0; i < MAXPOS; i++) {
        currentPassengers += diff[i];
        if (currentPassengers > capacity) {
            return false;
        }
    }
    
    return true;
}
```

## 优化技巧

### 1. 延迟计算

只在需要查询时才还原数组：

```cpp
class LazyDifference {
private:
    vector<long long> diff;
    mutable vector<long long> arr;
    mutable bool computed;
    int n;
    
    void compute() const {
        if (computed) return;
        
        arr[0] = diff[0];
        for (int i = 1; i < n; i++) {
            arr[i] = arr[i-1] + diff[i];
        }
        computed = true;
    }
    
public:
    LazyDifference(int size) : n(size), computed(false) {
        diff.resize(n + 1, 0);
        arr.resize(n);
    }
    
    void rangeAdd(int l, int r, int val) {
        diff[l] += val;
        if (r + 1 < n) diff[r + 1] -= val;
        computed = false;
    }
    
    long long get(int idx) const {
        compute();
        return arr[idx];
    }
};
```

### 2. 压缩坐标

当坐标范围很大但操作较少时：

```cpp
struct Operation {
    int l, r, val;
};

vector<long long> compressedDifference(vector<Operation>& ops, int maxPos) {
    vector<int> coords;
    coords.push_back(0);
    coords.push_back(maxPos);
    
    for (const auto& op : ops) {
        coords.push_back(op.l);
        coords.push_back(op.r);
        coords.push_back(op.r + 1);
    }
    
    sort(coords.begin(), coords.end());
    coords.erase(unique(coords.begin(), coords.end()), coords.end());
    
    int n = coords.size();
    vector<long long> diff(n, 0);
    
    for (const auto& op : ops) {
        int l = lower_bound(coords.begin(), coords.end(), op.l) - coords.begin();
        int r = lower_bound(coords.begin(), coords.end(), op.r + 1) - coords.begin();
        
        diff[l] += op.val;
        if (r < n) diff[r] -= op.val;
    }
    
    // 还原压缩后的数组
    for (int i = 1; i < n; i++) {
        diff[i] += diff[i-1];
    }
    
    return diff;
}
```

## 差分与其他算法的结合

### 1. 差分 + 二分

```cpp
// 判断能否通过k次操作使所有元素相等
bool canMakeEqual(vector<int>& arr, int k) {
    int n = arr.size();
    vector<int> diff(n + 1, 0);
    
    // 计算需要的操作次数
    int operations = 0;
    for (int i = 0; i < n - 1; i++) {
        if (arr[i] > arr[i + 1]) {
            int need = arr[i] - arr[i + 1];
            diff[0] += need;
            diff[i + 1] -= need;
            operations += need;
        }
    }
    
    return operations <= k;
}
```

### 2. 差分 + 贪心

```cpp
// 最少操作次数使数组非递减
int minOperations(vector<int>& arr) {
    int n = arr.size();
    int operations = 0;
    
    for (int i = n - 2; i >= 0; i--) {
        if (arr[i] > arr[i + 1]) {
            operations += arr[i] - arr[i + 1];
            arr[i] = arr[i + 1];
        }
    }
    
    return operations;
}
```

## 注意事项

1. **边界处理**：注意差分数组的边界，避免数组越界
2. **数据类型**：根据数据范围选择合适的数据类型，防止溢出
3. **操作顺序**：所有修改操作完成后再进行还原操作
4. **负数处理**：模运算时要正确处理负数

## 总结

差分算法是处理区间修改问题的利器，特别适用于：
- 大量区间修改操作
- 最后统一查询结果
- 需要高效处理区间加法的场景

掌握一维和二维差分的基本原理和实现，能够有效解决各种区间修改问题。差分算法的核心思想是将区间修改转化为边界标记，从而实现高效的批量操作。
